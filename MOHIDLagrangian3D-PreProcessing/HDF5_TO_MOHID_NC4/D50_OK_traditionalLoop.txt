# Combine rugosity and fractions in lists
rug_values_with_bkg = rug_values + [bkg_rugosity]
fractions_with_bkg = fractions + [f_bkg]

# Stack only non-zero layers
rug_stack = xr.concat([xr.DataArray(r) for r in rug_values_with_bkg], dim="layer")
frac_stack = xr.concat([xr.DataArray(f) for f in fractions_with_bkg], dim="layer")

# Sort by rugosity ascending
sort_idx = xr.DataArray(np.argsort(rug_stack.values), dims="layer")
rug_sorted = rug_stack.isel(layer=sort_idx)
frac_sorted = frac_stack.isel(layer=sort_idx)

# Prepare final D50 array
D50_val = xr.full_like(frac_sorted.isel(layer=0), fill_value=fillValue)

# Loop over each cell
for lat_index, lat_val in enumerate(frac_sorted.lat.values):
    for lon_indx, lon_val in enumerate(frac_sorted.lon.values):
        rug_cell = []
        frac_cell = []
        f_bkg_cell = []
        for layer_indx, layer_val in enumerate(rug_sorted.values):
            # Extract fractions and rugosity per cell
            rug_cell.append(layer_val.item())
            frac_cell_val = frac_sorted.isel(layer=layer_indx, lat=lat_index, lon=lon_indx).item()
            frac_cell.append(frac_cell_val)
            frac_cell_cumulative = np.cumsum(frac_cell)
            f_bkg_cell = f_bkg.isel(lat=lat_index, lon=lon_indx).item()

        # First index where cumulative fraction >= 0.5
        first_idx = np.argmax(frac_cell_cumulative > 0.5)

        # Find previous index with fraction > 0
        prev_idx = first_idx - 1
        while prev_idx > 0 and frac_cell[prev_idx] < 1e-4:
            prev_idx -= 1
        if frac_cell[prev_idx] == 0:
            prev_idx = first_idx

        # Values for interpolation
        D1 = rug_cell[prev_idx]
        D2 = rug_cell[first_idx]
        f1 = frac_cell_cumulative[prev_idx]
        f2 = frac_cell_cumulative[first_idx]

        # Interpolate only if 0.5 lies strictly between f1 and f2 and first_idx > 0
        if (f1 < 0.5) and (f2 > 0.5) and (first_idx > 0) and (f_bkg_cell != 1.0):
            if rugosity_method == 'Folk_PowAvg':
                D50_cell = np.exp(np.log(D1) + (0.5 - f1) / (f2 - f1) * (np.log(D2) - np.log(D1)))
            else:
                D50_cell = D1 + (0.5 - f1) / (f2 - f1) * (D2 - D1)
        else:
            D50_cell = D2 if f_bkg_cell != 1.0 else bkg_rugosity

        # Assign to D50 array
        D50_val.loc[dict(lat=lat_val, lon=lon_val)] = D50_cell

    # Mask and finalize
    final_D50 = D50_val.where(fraction_sum >= 0, other=fillValue)
    final_D50 = final_D50.where(~sea_mask, other=background_rugosity)
    final_D50.name = "D50"
    ds["D50"] = final_D50
